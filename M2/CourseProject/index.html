Module 2

Course Project: Developing Logic

Task: Write pseudocode to prompt for two numbers, display results of sum, product, and tripled sum functions

User Input

Declare n1 <- undefined
Declare n2 <- undefined

Input n1, n2

n1 <- parse n1 into number
n2 <- parse n2 into number

If n1 is not a number or n2 is not a number
	Return Display Error 
End If

Sum Function

Call Sum (n1, n2)

Function Sum (n1 number, n2 number)
	Declare result <- n1 + n2
	Declare target <- DOM Element with id "sum-target"
	Set target innerHTML <- result
End Function

Product Function

Call Product (n1, n2)

Function Product (n1 number, n2 number)
	Declare result <- n1 * n2
	Declare target <- DOM Element with id "product-target"
	Set target innerHTML <- result
End Function

Tripled Sum Function

Call TripledSum (n1, n2)

Function TripledSum (n1 number, n2 number)
	Declare result <- (n1 + n2) * 3
	Declare target <- DOM Element with id "tripledsum-target"
	Set target innerHTML <- result
End Function
